"def create_temp_dir(prefix, inner_dir=None, tmp_dir=None):    """    Create intermediate dirs <tmp>/<prefix+random suffix>/<inner_dir>/    argument 'tmp_dir' is used in unit tests    """    if not tmp_dir:        tmp_dir_path = tempfile.mkdtemp(prefix=prefix)    else:        tmp_dir_path = tempfile.mkdtemp(prefix=prefix, dir=tmp_dir)    if inner_dir:        tmp_dir_path = os.path.join(tmp_dir_path, inner_dir)        os.mkdir(tmp_dir_path, 0o700)    return tmp_dir_path"
