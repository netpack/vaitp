"    def values_from_list(values: List[str]) -> Dict[str, str]:        """ Convert a list of values into dict of {name: value} """        output: Dict[str, str] = {}        for i, value in enumerate(values):            if value[0].isalpha():                key = value.upper()            else:                key = f"VALUE_{i}"            if key in output:                raise ValueError(f"Duplicate key {key} in Enum")            output[key] = value        return output"
