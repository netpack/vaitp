Module(
    body=[
        ImportFrom(
            module='ecdsa',
            names=[
                alias(name='VerifyingKey'),
                alias(name='BadSignatureError')],
            level=0),
        Assign(
            targets=[
                Name(id='public_key', ctx=Store())],
            value=Constant(value='-----BEGIN PUBLIC KEY-----\n...\n-----END PUBLIC KEY-----')),
        Assign(
            targets=[
                Name(id='malformed_signature', ctx=Store())],
            value=Constant(value=b'malformed_signature_data')),
        Assign(
            targets=[
                Name(id='vk', ctx=Store())],
            value=Call(
                func=Attribute(
                    value=Name(id='VerifyingKey', ctx=Load()),
                    attr='from_pem',
                    ctx=Load()),
                args=[
                    Name(id='public_key', ctx=Load())],
                keywords=[])),
        Try(
            body=[
                Expr(
                    value=Call(
                        func=Attribute(
                            value=Name(id='vk', ctx=Load()),
                            attr='verify',
                            ctx=Load()),
                        args=[
                            Name(id='malformed_signature', ctx=Load()),
                            Constant(value=b'message')],
                        keywords=[]))],
            handlers=[
                ExceptHandler(
                    type=Name(id='BadSignatureError', ctx=Load()),
                    body=[
                        Expr(
                            value=Call(
                                func=Name(id='print', ctx=Load()),
                                args=[
                                    Constant(value='Signature verification failed.')],
                                keywords=[]))]),
                ExceptHandler(
                    type=Name(id='Exception', ctx=Load()),
                    name='e',
                    body=[
                        Expr(
                            value=Call(
                                func=Name(id='print', ctx=Load()),
                                args=[
                                    JoinedStr(
                                        values=[
                                            Constant(value='An unexpected error occurred: '),

                                                value=Name(id='e', ctx=Load()),
                                                conversion=-1)])],
                                keywords=[]))])],
            orelse=[],
            finalbody=[])],
    type_ignores=[])
